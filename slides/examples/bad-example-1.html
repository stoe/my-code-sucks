function foo (x) {
    if (x == 1) {
        console.log('%s + 5 = %s', x, (x + 5));
    } else {
        console.log('NaN');
    }
}


foo(1);
foo('1');


// => 1 + 5 = 6
// => '1' + 5 = 15


foo(true);
foo(false);


// => true + 5 = 6
// => NaN


foo([]);
foo([1]);


// => NaN
// => Array[1] + 5 = 15


foo({});
foo({bar: 1});


// => NaN
